# 5. Codigo

from turtle import *

from freegames import vector

def line(start, end):
  """Draw line from start to end."""
  up()
  goto(start.x, start.y)
  down()
  goto(end.x, end.y)

def square(start, end):
  """Draw square from start to end."""
  up()
  goto(start.x, start.y)
  down()
  begin_fill()

for count in range(4):
  forward(end.x - start.x)
  left(90)

end_fill()
